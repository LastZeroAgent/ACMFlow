def two_sum(nums, target):
    n = len(nums)
    for i in range(n):
        for j in range(i + 1, n):
            if nums[i] + nums[j] == target:
                return [i, j]
    return []

方法二：哈希表法（优化）

Python实现
def two_sum(nums, target):
    num_map = {}
    for i, num in enumerate(nums):
        complement = target - num
        if complement in num_map:
            return [num_map[complement], i]
        num_map[num] = i
    return []

n = int(input())

C++实现
#include <iostream>
#include <vector>
#include <unordered_map>
using namespace std;

vector<int> twoSum(vector<int>& nums, int target) {
    unordered_map<int, int> numMap;
    for (int i = 0; i < nums.size(); ++i) {
        int complement = target - nums[i];
        if (numMap.find(complement) != numMap.end()) {
            return {numMap[complement], i};
        }
        numMap[nums[i]] = i;
    }
    return {};
}

import java.util.HashMap;

        for (int i = 0; i < nums.length; ++i) {

题解分析框架（两数之和）

<solution>
## 题解：两数之和

### 方法一：暴力枚举法

#### 算法思路
暴力枚举法是最直接的解法。我们遍历数组中的每一个元素，并检查是否存在另一个元素与它相加等于目标值。具体来说：

1. 外层循环遍历数组中的每一个元素
2. 内层循环从外层循环的下一个元素开始遍历
3. 如果找到两个元素的和等于目标值，就返回它们的下标

这种方法的思路简单直观，但效率相对较低。

#### 复杂度分析
- **时间复杂度**：O(n²)，其中n是数组的长度。最坏情况下，我们需要检查数组中所有可能的元素对。
- **空间复杂度**：O(1)，只使用了常数级别的额外空间。

#### 代码解释
- `n = len(nums)` 获取数组长度
- 双重循环遍历所有可能的元素对
- 条件判断`nums[i] + nums[j] == target`检查是否找到解
- 找到后立即返回结果

### 方法二：哈希表法（优化）

#### 算法思路
哈希表法通过使用额外的空间来换取时间效率的提升。具体思路是：

1. 创建一个哈希表来存储元素值到下标的映射
2. 遍历数组中的每个元素
3. 对于每个元素nums[i]，计算补数`complement = target - nums[i]`
4. 检查补数是否在哈希表中
5. 如果存在，说明找到了两个数，返回哈希表中补数的下标和当前下标
6. 如果不存在，将当前元素及其下标存入哈希表

这种方法只需要遍历数组一次，时间效率显著提升。

#### 复杂度分析
- **时间复杂度**：O(n)，其中n是数组的长度。我们只需要遍历数组一次。
- **空间复杂度**：O(n)，最坏情况下需要存储数组中的所有元素。

#### 代码解释
- `num_map = {}` 创建一个空的哈希表
- 遍历数组的同时获取元素值和下标：`for i, num in enumerate(nums)`
- 计算补数：`complement = target - num`
- 检查补数是否在哈希表中：`if complement in num_map`
- 如果存在，返回哈希表中存储的下标和当前下标
- 如果不存在，将当前元素和下标存入哈希表：`num_map[num] = i`

### 总结
两数之和问题有两种主要解法：
1. 暴力枚举法：简单直观，时间复杂度O(n²)
2. 哈希表法：高效优化，时间复杂度O(n)

在实际应用中，当数组长度较大时，哈希表法的优势会非常明显。建议掌握并使用哈希表法作为首选解决方案。
</solution>

以下是一个简化的Python实现，用于自动验证算法代码： 

import time
import sys
import resource
import subprocess

def run_test_case(code, test_case):
    """运行单个测试案例"""
    try:
        # 设置资源限制
        resource.setrlimit(resource.RLIMIT_CPU, (1, 1))  # 1秒时间限制
        resource.setrlimit(resource.RLIMIT_AS, (128 * 1024 * 1024, 128 * 1024 * 1024))  # 128MB空间限制
        
        # 准备输入数据
        input_data = test_case['input']
        
        # 运行代码
        start_time = time.time()
        process = subprocess.Popen(
            [sys.executable, '-c', code],
            stdin=subprocess.PIPE,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE
        )
        
        # 输入数据并获取输出
        output, error = process.communicate(input=input_data.encode())
        
        # 计算运行时间
        runtime = time.time() - start_time
        
        # 检查输出
        expected_output = test_case['output'].strip()
        actual_output = output.decode().strip()
        
        return {
            'passed': actual_output == expected_output,
            'actual_output': actual_output,
            'expected_output': expected_output,
            'runtime': runtime,
            'error': error.decode()
        }
    except Exception as e:
        return {
            'passed': False,
            'error': str(e),
            'runtime': -1
        }

def validate_algorithm(code, test_cases):
    """验证整个算法"""
    results = []
    total_passed = 0
    
    for i, test_case in enumerate(test_cases):
        print(f"正在运行测试用例 {i+1}...")
        result = run_test_case(code, test_case)
        results.append(result)
        
        if result['passed']:
            total_passed += 1
            print(f"测试用例 {i+1}: ✅ 通过 (运行时间: {result['runtime']:.4f}秒)")
        else:
            print(f"测试用例 {i+1}: ❌ 失败 (运行时间: {result['runtime']:.4f}秒)")
            print(f"  预期输出: {test_case['output'].strip()}")
            print(f"  实际输出: {result['actual_output']}")
            if result['error']:
                print(f"  错误信息: {result['error']}")
    
    print(f"\\n验证结果总结:")
    print(f"通过测试用例数: {total_passed}/{len(test_cases)}")
    print(f"通过率: {total_passed/len(test_cases)*100:.1f}%")
    
    return results

现在让我们尝试优化版本——哈希表法。这种方法的时间复杂度是O(n)，比暴力法的O(n²)更高效。

def two_sum(nums, target):
    num_map = {}
    for i, num in enumerate(nums):
        complement = target - num
        if complement in num_map:
            return [num_map[complement], i]
        num_map[num] = i
    return []

# 读取输入
n = int(input())
nums = list(map(int, input().split()))
target = int(input())

# 调用函数
result = two_sum(nums, target)
print(result[0], result[1])

哈希表法的核心思想是：

  1. 创建一个字典存储数值到下标的映射
  2. 遍历数组时，计算补数（target - num）
  3. 检查补数是否在字典中
  4. 如果存在，返回结果
  5. 否则，将当前数存入字典

这种方法只需要遍历一次数组，效率更高。
